# === INSTALLATION & RUN INSTRUCTIONS ===

# 1. Install backend dependencies:
pip install -r requirements.txt

# 2. Install frontend dependencies:
npm install

# 3. Start the backend (example, if using FastAPI):
# (You may need to adjust the command based on your entrypoint)
uvicorn backend.main:app --reload

# 4. Start the frontend (Vite):
npm run dev

# 5. (Optional) Start Storybook for UI components:
npm run storybook

# 6. Build frontend for production:
npm run build

# 7. Run frontend preview:
npm run preview

# 8. Run tests (if needed):
npm run test

# Make sure to configure environment variables and database connections as needed for your setup.
